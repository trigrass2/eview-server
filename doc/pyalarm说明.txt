#-*-函数名固定，字典键值固定7个，SQL拼接是个例子，勿模仿！！！-*-
def OnAlarm(alarm):  
    sql=''
    sql+=alarm["ProduceTime"]  
    sql+=alarm["Priority"] 
    sql+=alarm["Subsys"]  
    sql+=alarm["AlarmName"] 
    sql+=alarm["AlarmCName"]
    sql+=alarm["AlarmValue"]
    sql+=alarm["DeviceName"]
    return sql

#-*-调用例子-*-
OnAlarm(ProduceTime='A',Priority='B',...)

#-*-程序运行需将定义的py函数OnAlarm(alarm)文件copy到config文件夹下-*-
